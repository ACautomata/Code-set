#include<bits/stdc++.h>
using namespace std;
int dp[1010][1010],a[1010][1010],n,m;
int main(){
memset(dp,0,sizeof(dp));
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            cin>>a[i][j];
    
    for(int i=1;i<=n;i++)
        for(int j=0;j<=m;j++)    
            for(int k=0;k<=j;k++)
                    dp[i][j]=max(dp[i][j],dp[i-1][j-k]+a[i][k]);
    cout<<dp[n][m]<<endl;
    return 0;
}
